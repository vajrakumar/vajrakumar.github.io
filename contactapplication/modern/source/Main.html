/**

    alias: 'widget.app-main',

        'Ext.tab.Panel',
        'Ext.Toast'

    viewModel: 'contact',

    layout: {
        type: 'fit'
    },

    fullscreen: true,
    padding: 0,

    items: [

            xtype: 'contact-list'

        {
            xtype: 'contact-details',
